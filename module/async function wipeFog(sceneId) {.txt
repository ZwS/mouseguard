async function wipeFog(sceneId) {
  const fog = await FogExploration.get({scene: game.scenes.get(sceneId)});
  if ( !fog ) return;
  await fog.delete();
  await canvas.sight.loadFog();
}
socket.on("world", data => {
  if ( data.action === "wipeFog" ) return wipeFog(data.sceneId);
});

socket.emit("world", {action: "wipeFog", sceneId: canvas.scene.id}, {recipients: [user1.id, user2.id]});
